@app.route('/api/analyze', methods=['POST'])
def analyze():
    try:
        symptoms = json.loads(request.form.get('symptoms', '{}'))
        if 'audio' not in request.files:
            return jsonify(demo_analysis(symptoms))
        audio_file = request.files['audio']
        with tempfile.NamedTemporaryFile(suffix='.webm', delete=False) as tmp:
            audio_file.save(tmp.name)
            tmp_path = tmp.name
        try:
            waveform, sample_rate = torchaudio.load(tmp_path)
        except Exception:
            os.unlink(tmp_path)
            return jsonify(demo_analysis(symptoms))
        os.unlink(tmp_path)
        if waveform.shape[0] > 1:
            waveform = waveform.mean(dim=0, keepdim=True)
        if sample_rate != 16000:
            waveform = torchaudio.transforms.Resample(sample_rate, 16000)(waveform)
        audio_np = waveform.squeeze().numpy()
        features = extract_features(audio_np, 16000)
        hear_score = None
        if hear_model is not None:
            _, hear_score = run_hear(waveform)
        knn   = knn_classify(features)
        score = compute_final_score(features, knn, symptoms, hear_score)
        return jsonify({
            'features': features, 'knn': knn,
            'hear_score': hear_score, 'hear_available': hear_model is not None,
            'final_score': score, 'mode': 'hear+knn' if hear_model else 'knn'
        })
    except Exception as e:
        print(f"Error: {e}")
        return jsonify(demo_analysis(json.loads(request.form.get('symptoms', '{}'))))




def extract_features(audio, sr):
    N = len(audio)
    fft = np.fft.rfft(audio)
    freqs = np.fft.rfftfreq(N, 1/sr)
    mag = np.abs(fft)
    rms = float(np.sqrt(np.mean(audio**2)))
    zcr = float(np.sum(np.diff(np.sign(audio)) != 0) / N)
    sc  = float(np.sum(freqs * mag) / (np.sum(mag) + 1e-8))
    bw  = float(np.sqrt(np.sum(((freqs - sc)**2) * mag) / (np.sum(mag) + 1e-8)))
    cumsum = np.cumsum(mag)
    rolloff_idx = np.searchsorted(cumsum, 0.85 * cumsum[-1])
    ro = float(freqs[min(rolloff_idx, len(freqs)-1)])
    frame_size = max(N // 10, 1)
    energies = [float(np.sqrt(np.mean(audio[i*frame_size:(i+1)*frame_size]**2)))
                for i in range(10) if (i+1)*frame_size <= N]
    ev = float(np.var(energies)) if energies else 0.0
    cough_count = 0
    in_cough = False
    threshold = rms * 2.2
    for e in energies:
        if e > threshold and not in_cough:
            cough_count += 1; in_cough = True
        elif e < threshold * 0.6:
            in_cough = False
    return {
        'rms': round(min(rms, 1.0), 4), 'zcr': round(min(zcr, 1.0), 4),
        'sc': int(min(sc, 4300)), 'bw': int(min(bw, 3200)),
        'ro': int(min(ro, 7800)), 'ev': round(min(ev, 1.0), 4),
        'coughCount': max(1, cough_count), 'duration': round(N / 16000, 1),
    }




def run_hear(waveform):
    try:
        import tensorflow as tf
        target = 32000
        audio = waveform.squeeze().numpy()
        if len(audio) < target:
            audio = np.pad(audio, (0, target - len(audio)))
        else:
            audio = audio[:target]
        audio_tf  = tf.constant(audio[np.newaxis, :], dtype=tf.float32)
        embedding = hear_model.signatures["serving_default"](audio_tf)
        emb = list(embedding.values())[0].numpy().squeeze()
        irregularity   = float(np.std(emb))
        low_dim_energy = float(np.mean(np.abs(emb[:128])))
        hi_dim_energy  = float(np.mean(np.abs(emb[128:])))
        hear_score = min(40, int(irregularity * 20 + low_dim_energy / (hi_dim_energy + 1e-8) * 5))
        return emb.tolist(), hear_score
    except Exception as e:
        print(f"HeAR error: {e}")
        return None, None




def knn_classify(features, k=5):
    norm = {
        'sc': min(features['sc']/4300, 1.0), 'bw': min(features['bw']/3200, 1.0),
        'zcr': min(features['zcr'], 1.0),    'rms': min(features['rms'], 1.0),
        'ro': min(features['ro']/7800, 1.0), 'ev': min(features['ev']/0.5, 1.0),
    }
    distances = []
    for ref in REFERENCE_LIBRARY:
        rn = {'sc': ref['sc']/4300, 'bw': ref['bw']/3200, 'zcr': ref['zcr'],
              'rms': ref['rms'], 'ro': ref['ro']/7800, 'ev': ref['ev']/0.5}
        d = (2.0*(norm['sc']-rn['sc'])**2 + 1.5*(norm['zcr']-rn['zcr'])**2 +
             1.5*(norm['rms']-rn['rms'])**2 + 1.0*(norm['bw']-rn['bw'])**2 +
             0.8*(norm['ro']-rn['ro'])**2 + 0.7*(norm['ev']-rn['ev'])**2) ** 0.5
        distances.append({**ref, 'distance': round(d, 4)})
    distances.sort(key=lambda x: x['distance'])
    k_nearest = distances[:k]
    votes = {}
    total_w = 0
    for n in k_nearest:
        w = 1 / (n['distance'] + 0.001)
        votes[n['label']] = votes.get(n['label'], 0) + w
        total_w += w
    top_label  = max(votes, key=votes.get)
    confidence = round(votes[top_label] / total_w * 100, 1)
    return {
        'kNearest': k_nearest, 'topLabel': top_label, 'confidence': confidence,
        'votes': {k: round(v/total_w*100, 1) for k, v in votes.items()},
    }




def compute_final_score(features, knn, symptoms, hear_score=None):
    label_risk = {"Healthy": 0, "Dry/Mild Irritation": 25,
                  "Obstructive Pattern": 55, "Wet/Productive": 60,
                  "Restrictive Pattern": 65, "Symptomatic": 70}
    acoustic = sum((pct/100) * label_risk.get(label, 0) for label, pct in knn['votes'].items())
    if hear_score is not None:
        acoustic = (acoustic * 0.7) + (hear_score * 0.3 * 70/40)
    acoustic = min(round(acoustic), 70)
    age   = int(symptoms.get('age',   30))
    smoke = int(symptoms.get('smoke',  0))
    years = int(symptoms.get('years',  0))
    cough = int(symptoms.get('cough',  0))
    sob   = int(symptoms.get('sob',    0))
    chest = int(symptoms.get('chest',  0))
    cook  = int(symptoms.get('cook',   0))
    sym  = 0
    sym += 8 if age > 55 else 5 if age > 45 else 2 if age > 35 else 0
    sym += smoke * 4 + min(years * 0.4, 5) + cough * 4 + sob * 4 + chest * 4 + cook * 2
    return min(acoustic + min(int(sym), 30), 100)




def demo_analysis(symptoms):
    features = {'rms': 0.48, 'zcr': 0.22, 'sc': 2600,
                'bw': 2000, 'ro': 5200, 'ev': 0.20,
                'coughCount': 3, 'duration': 5.0}
    knn   = knn_classify(features)
    score = compute_final_score(features, knn, symptoms)
    return {'features': features, 'knn': knn, 'hear_score': None,
            'hear_available': False, 'final_score': score, 'mode': 'demo'}
